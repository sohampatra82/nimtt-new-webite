    <script>
        const hamburger = document.getElementById("hamburger");
        const menu = document.querySelector(".menu");
        let menuOpen = false;

        // Smooth Hamburger toggle using GSAP
        hamburger.addEventListener("click", () => {
            if (!menuOpen) {
                menu.classList.add("active");
                gsap.fromTo(menu,
                    { opacity: 0, y: -20 },
                    { opacity: 1, y: 0, duration: 0.4, ease: "power2.out" }
                );
            } else {
                gsap.to(menu, {
                    opacity: 0,
                    y: -20,
                    duration: 0.4,
                    ease: "power2.in",
                    onComplete: () => {
                        menu.classList.remove("active");
                    }
                });
            }
            menuOpen = !menuOpen;
        });

        // Logo animation
        gsap.from(".navbar .logo img", {
            duration: 1,
            x: -50,
            opacity: 0,
            ease: "power3.out"
        });

        // Menu items animation (on initial load only)
        gsap.from(".menu li", {
            duration: 1,
            y: -20,
            opacity: 0,
            stagger: 0.1,
            ease: "power2.out",
            delay: 0.3
        });
    </script>
